\begin{denotation}[3cm]
  \item[ALPT] 增强拉格朗日微扰论（Augmented Lagrangian Perturbation Theory）
  \item[CMB] 宇宙微波背景辐射（Cosmic Microwave Background）
  \item[CSST] 中国空间站望远镜（Chinese Space Station Telescope）
  \item[DESI] 暗能量光谱巡天（Dark Energy Spectroscopic Instrument）
  \item[DR] 数据发布（Data Release）
  \item[GAN] 生成对抗网络（Generative Adversarial Network）
  \item[PDF] 概率分布函数（Probability Density Function）
  \item[SDSS] 斯隆数字巡天（Sloan Digital Sky Survey） 
  \item[SZ 效应] Sunyaev-Zel'dovich 效应
  \item[WGAN-GP] 带梯度惩罚的 Wasserstein 生成对抗网络（Wasserstein GAN with Gradient Penalty）
  \item[ZA] Zel'dovich 近似
  \item[2LPT] 二阶拉格朗日微扰论（Second-order Lagrangian Perturbation Theory）
  \item[$\Lambda$CDM] 冷暗物质模型（Lambda Cold Dark Matter）
  \item[$\delta$] 暗物质过密度场
  \item[$\rho$] 暗物质密度
  \item[$\bar{\rho}$] 平均暗物质密度
  \item[$\boldsymbol{\Psi}$] 粒子位移场
  \item[$\boldsymbol{v}$] 粒子速度场
  \item[$D(z)$] 线性增长因子
  \item[$f(z)$] 线性增长率
  \item[$H(z)$] 红移 $z$ 处的哈勃参数
  \item[$h$] 无量纲哈勃参数
  \item[$P(k)$] 暗物质功率谱
  \item[$B(k_1,k_2,k_3)$] 暗物质双谱
\end{denotation}
